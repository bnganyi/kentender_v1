**6\. Tests**

**Test 1 — Empty state**

**File**

budget-empty.spec.ts

import { test, expect } from '@playwright/test';

import { loginAsStrategyManager } from '../helpers/auth';

import { openBudgetLanding } from '../helpers/budget';

test('Budget landing shows empty state when no budgets exist', async ({ page }) => {

await loginAsStrategyManager(page);

await openBudgetLanding(page);

await expect(page.getByTestId('budget-empty-state'))

.toContainText('No budgets yet. Create one to begin.');

await expect(page.getByTestId('budget-create-button')).toBeVisible();

});

**Test 8 — Allocation save**

test('Allocation can be created or updated', async ({ page }) => {

await loginAsStrategyManager(page);

await openBudgetLanding(page);

await page.getByTestId('selected-budget-open-builder').click();

await page.getByTestId('budget-program-row-Healthcare Access').click();

const input = page.getByTestId('budget-allocation-amount-input');

await input.fill('5500000');

await page.getByTestId('budget-allocation-save-button').click();

.toContainText('5,500,000');

});
